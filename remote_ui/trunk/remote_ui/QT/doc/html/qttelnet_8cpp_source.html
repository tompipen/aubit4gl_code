00608 void QtTelnetPrivate::consume()
00609 {
00610     const QByteArray data = buffer.readAll();
00611     int currpos = 0;
00612     int prevpos = -1;;
00613     while (prevpos < currpos && currpos < data.size()) {
00614         prevpos = currpos;
00615         const uchar c = uchar(data[currpos]);
00616         if (c == Common::DM)
00617             ++currpos;

00619             currpos += parseIAC(data.mid(currpos));

00621             currpos += parsePlaintext(data.mid(currpos));
00622     }
00623     if (currpos < data.size())
00624         buffer.push_back(data.mid(currpos));

00713 /*
00714   returns number of bytes consumed
00715 */
00716 int QtTelnetPrivate::parseIAC(const QByteArray &data)
00717 {
00718     if (data.isEmpty())
00719         return 0;
00720 
00721     Q_ASSERT(uchar(data.at(0)) == Common::IAC);
00722 
00723     if (data.size() >= 3 && isOperation(data[1])) { // IAC, Operation, Option
00724         const uchar operation = data[1];
00725         const uchar option = data[2];
00726         if (operation == Common::WONT && option == Common::Logout) {
00727             q->close();
00728             return 3;
00729         }
00730         if (operation == Common::DONT && option == Common::Authentication) {
00731             if (loginp.isEmpty() && passp.isEmpty())
00732                 emit q->loggedIn();
00733             nullauth = true;
00734         }
00735         if (replyNeeded(operation, option)) {
00736             bool allowed = allowOption(operation, option);
00737             sendCommand(opposite(operation, allowed), option);
00738             setMode(operation, option);
00739         }
